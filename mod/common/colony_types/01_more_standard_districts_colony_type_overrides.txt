# Tech-World (Research)
col_research = {
	icon = GFX_colony_type_research
	potential = {
		msd_uses_district_set_city_world = no
		msd_uses_district_set_habitat = no
		msd_uses_district_set_ring_world = no
		is_special_colony_type = no
	}
	planet_modifier = {
		planet_building_research_lab_1_build_speed_mult = 0.25
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_technology = tech_basic_science_lab_2 }
		planet_building_research_lab_2_build_speed_mult = 0.25
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_technology = tech_basic_science_lab_3 }
		planet_building_research_lab_3_build_speed_mult = 0.25
		}
	}
	triggered_planet_modifier = {
		potential = { msd_uses_regular_district_set = yes }
		planet_district_standard_cm_science_build_speed_mult = 0.25
	}
	triggered_planet_modifier = {
		potential = { always = yes }
		# always-on triggered modifier for ordering purposes
		planet_researchers_upkeep_mult = -0.2
	}
	weight_modifier = {
		base = 1
		modifier = {
			add = 5
			has_designation = col_research
		}
		modifier = { add = value:col_science_composite_dynamic_weight }
		modifier = {
			factor = 0
			exists = sector
			sector = {
				has_any_sector_focus = yes
				NOR = {
					has_sector_focus = "research"
					has_sector_focus = "balanced"
				}
			}
		}
	}
}

	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_technology = tech_paradise_dome }
		}
		planet_building_organic_paradise_build_speed_mult = 0.25
	}
	triggered_planet_modifier = {
		potential = { msd_uses_regular_district_set = yes }
		planet_district_standard_cm_admin_machine_build_speed_mult = 0.25
	}
	weight_modifier = {
		base = 10
		modifier = {
			add = 5
			has_designation = col_trophy_machine
		}
		modifier = {
			add = 3
			OR = {
				num_buildings = { type = building_organic_sanctuary value >= 2 }
				num_districts = { type = district_arcology_organic_housing value >= 2 }
			}
		}
		modifier = {
			add = 4
			OR = {
				num_buildings = { type = building_organic_sanctuary value >= 3 }
				num_districts = { type = district_arcology_organic_housing value >= 3 }
			}
		}
		modifier = {
			add = 5
			OR = {
				num_buildings = { type = building_organic_sanctuary value >= 4 }
				num_districts = { type = district_arcology_organic_housing value >= 4 }
			}
		}
		modifier = {
			add = 6
			OR = {
				num_buildings = { type = building_organic_sanctuary value >= 5 }
				num_districts = { type = district_arcology_organic_housing value >= 5 }
			}
		}
		modifier = {
			add = 5
			num_buildings = { type = building_organic_paradise value >= 1 }
		}
		modifier = {
			add = 6
			num_buildings = { type = building_organic_paradise value >= 2 }
		}
		modifier = {
			factor = 0
			AND = {
				num_buildings = { type = building_organic_sanctuary value = 0 }
				num_buildings = { type = building_organic_paradise value = 0 }
				num_districts = { type = district_arcology_organic_housing value = 0 }
			}
		}
		modifier = {
			factor = 0
			exists = sector
			sector = {
				has_any_sector_focus = yes
				not = { has_sector_focus = "unity" }

			}
		}
		modifier = {
			add = 1
			exists = sector
			sector = {
				has_sector_focus = "unity"
			}
		}
	}
